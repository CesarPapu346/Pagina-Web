  <script>
    let order = {};
    let totalPrice = 0;

    function addToOrder() {
      const platilloSelect = document.getElementById('platillo');
      const cantidadInput = document.getElementById('cantidad');
      const platillo = platilloSelect.value;
      const cantidad = parseInt(cantidadInput.value);
      const price = parseFloat(platilloSelect.selectedOptions[0].getAttribute('data-price'));

      if (order[platillo]) {
        order[platillo].cantidad += cantidad;
      } else {
        order[platillo] = { cantidad: cantidad, precio: price };
      }

      updateOrderSummary();
    }

    function updateOrderSummary() {
      const orderList = document.getElementById('orderList');
      const totalPriceElement = document.getElementById('totalPrice');
      orderList.innerHTML = '';
      totalPrice = 0;

      for (let platillo in order) {
        const item = order[platillo];
        const itemTotal = item.cantidad * item.precio;
        totalPrice += itemTotal;
        const li = document.createElement('li');
        li.textContent = `${platillo.charAt(0).toUpperCase() + platillo.slice(1)} x${item.cantidad} - $${itemTotal.toFixed(2)}`;
        orderList.appendChild(li);
      }

      totalPriceElement.textContent = totalPrice.toFixed(2);
    }
  </script>
